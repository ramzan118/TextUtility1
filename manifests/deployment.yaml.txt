apiVersion: apps/v1
kind: Deployment
metadata:
  name: textutility-app
  labels:
    app: textutility
spec:
  replicas: 3
  strategy:
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0
    type: RollingUpdate
  selector:
    matchLabels:
      app: textutility
  template:
    metadata:
      labels:
        app: textutility
    spec:
      containers:
      - name: textutility
        image: myimagerepo.azurecr.io/textutility:$(Build.BuildId)
        imagePullPolicy: Always
        ports:
        - containerPort: 80
        env:
        - name: DB_HOST
          valueFrom:
            secretKeyRef:
              name: db-secrets
              key: host
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: db-secrets
              key: password
        - name: COSMOS_CONNECTION_STRING
          valueFrom:
            secretKeyRef:
              name: db-secrets
              key: cosmos-conn
        livenessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 5
          periodSeconds: 10
      imagePullSecrets:
      - name: acr-auth

---
apiVersion: v1
kind: Service
metadata:
  name: textutility-service
spec:
  type: LoadBalancer
  ports:
  - port: 80
    targetPort: 80
  selector:
    app: textutility
